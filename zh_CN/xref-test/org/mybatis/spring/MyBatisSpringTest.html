400   @Test
401   void testSqlSessionCommitWithTx() {
402     var txDef = new DefaultTransactionDefinition();
403     txDef.setPropagationBehaviorName("PROPAGATION_REQUIRED");
404 
405     var status = txManager.getTransaction(txDef);
406 
407     session = SqlSessionUtils.getSqlSession(sqlSessionFactory);
408     session.getMapper(TestMapper.class).findTest();
409     // commit should no-op since there is an active transaction
410     session.commit(true);
411     SqlSessionUtils.closeSqlSession(session, sqlSessionFactory);
412 
413     txManager.commit(status);
414 
415     // Connection should be committed once, but we explicitly called commit on the SqlSession,
416     // so it should be committed twice
417     assertThat(connection.getNumberCommits()).as("should call commit on Connection").isEqualTo(1);
418     assertThat(connection.getNumberRollbacks()).as("should not call rollback on Connection").isEqualTo(0);
419     assertThat(executorInterceptor.getCommitCount()).as("should call commit on SqlSession").isEqualTo(2);
420     assertThat(executorInterceptor.getRollbackCount()).as("should not call rollback on SqlSession").isEqualTo(0);
421 
422     assertSingleConnection();
423   }

425   @Test
426   void testWithInterleavedTx() {
427     // this session will use one Connection
428     session = SqlSessionUtils.getSqlSession(sqlSessionFactory);
429     session.getMapper(TestMapper.class).findTest();
430 
431     // this transaction should use another Connection
432     var status = txManager.getTransaction(new DefaultTransactionDefinition());
433 
434     // session continues using original connection
435     session.getMapper(TestMapper.class).insertTest("test2");
436     session.commit(true);
437     SqlSessionUtils.closeSqlSession(session, sqlSessionFactory);
438 
439     // this should succeed
440     // SpringManagedTransaction (from SqlSession.commit()) should not interfere with tx
441     txManager.commit(status);
442 
443     // two transactions should have completed, each using their own Connection
444     assertThat(dataSource.getConnectionCount()).as("should call DataSource.getConnection() twice").isEqualTo(2);
445 
446     // both connections should be committed
447     assertThat(connection.getNumberCommits()).as("should call commit on Connection 1").isEqualTo(1);
448     assertThat(connection.getNumberRollbacks()).as("should not call rollback on Connection 1").isEqualTo(0);
449 
450     assertThat(connectionTwo.getNumberCommits()).as("should call commit on Connection 2").isEqualTo(1);
451     assertThat(connectionTwo.getNumberRollbacks()).as("should not call rollback on Connection 2").isEqualTo(0);
452 
453     // the SqlSession should have also committed and executed twice
454     assertCommitSession();
455     assertExecuteCount(2);
456 
457     assertConnectionClosed(connection);
458     assertConnectionClosed(connectionTwo);
459   }

461   @Test
462   void testSuspendAndResume() {
463 
464     try {
465       txManager.setDataSource(dataSource);
466       var status = txManager.getTransaction(new DefaultTransactionDefinition());
467 
468       session = SqlSessionUtils.getSqlSession(sqlSessionFactory);
469 
470       // start a new tx while the other is in progress
471       var txRequiresNew = new DefaultTransactionDefinition();
472       txRequiresNew.setPropagationBehaviorName("PROPAGATION_REQUIRES_NEW");
473       var status2 = txManager.getTransaction(txRequiresNew);
474 
475       var session2 = SqlSessionUtils.getSqlSession(sqlSessionFactory);
476 
477       assertThat(session).as("getSqlSession() should not return suspended SqlSession").isNotSameAs(session2);
478 
479       SqlSessionUtils.closeSqlSession(session2, sqlSessionFactory);
480       txManager.commit(status2);
481 
482       // first tx should be resumed now and this should succeed
483       session.getMapper(TestMapper.class).findTest();
484       SqlSessionUtils.closeSqlSession(session, sqlSessionFactory);
485       txManager.commit(status);
486 
487       // two transactions should have completed, each using their own Connection
488       assertThat(dataSource.getConnectionCount()).as("should call DataSource.getConnection() twice").isEqualTo(2);
489 
490       // both connections and should be committed
491       assertThat(connection.getNumberCommits()).as("should call commit on Connection 1").isEqualTo(1);
492       assertThat(connection.getNumberRollbacks()).as("should not call rollback on Connection 1").isEqualTo(0);
493 
494       assertThat(connectionTwo.getNumberCommits()).as("should call commit on Connection 2").isEqualTo(1);
495       assertThat(connectionTwo.getNumberRollbacks()).as("should not call rollback on Connection 2").isEqualTo(0);
496 
497       // the SqlSession should have also committed twice
498       assertThat(executorInterceptor.getCommitCount()).as("should call commit on SqlSession").isEqualTo(2);
499       assertThat(executorInterceptor.getRollbackCount()).as("should call rollback on SqlSession").isEqualTo(0);
500 
501       assertConnectionClosed(connection);
502       assertConnectionClosed(connectionTwo);
503     } finally {
504       // reset the txManager; keep other tests from potentially failing
505       txManager.setDataSource(dataSource);
506 
507       // null the connection since it was not used
508       // this avoids failing in validateConnectionClosed()
509       connection = null;
510     }
511   }
