360   @Test
361   void testScopedProxyMapperScanByDefaultScope() {
362     applicationContext.register(ScopedProxy.class);
363 
364     startContext();
365 
366     List<String> scopedProxyTargetBeans = Stream.of(applicationContext.getBeanDefinitionNames())
367         .filter(x -> x.startsWith("scopedTarget")).collect(Collectors.toList());
368     assertThat(scopedProxyTargetBeans).hasSize(1).contains("scopedTarget.ds1Mapper");
369 
370     for (String scopedProxyTargetBean : scopedProxyTargetBeans) {
371       {
372         var definition = applicationContext.getBeanDefinition(scopedProxyTargetBean);
373         assertThat(definition.getBeanClassName()).isEqualTo("org.mybatis.spring.mapper.MapperFactoryBean");
374         assertThat(definition.getScope()).isEqualTo("thread");
375       }
376       {
377         var definition = applicationContext.getBeanDefinition(scopedProxyTargetBean.substring(13));
378         assertThat(definition.getBeanClassName()).isEqualTo("org.springframework.aop.scope.ScopedProxyFactoryBean");
379         assertThat(definition.getScope()).isEqualTo("");
380       }
381     }
382     {
383       var mapper = applicationContext.getBean(Ds1Mapper.class);
384       assertThat(mapper.test()).isEqualTo("ds1");
385     }
386     {
387       var mapper = applicationContext.getBean(Ds2Mapper.class);
388       assertThat(mapper.test()).isEqualTo("ds2");
389     }
390     var sqlSessionFactory = applicationContext.getBean(SqlSessionFactory.class);
391     assertEquals(2, sqlSessionFactory.getConfiguration().getMapperRegistry().getMappers().size());
392   }
