16  package org.mybatis.spring.config;
17  
18  import java.lang.annotation.Annotation;
19  
20  import org.mybatis.spring.mapper.MapperFactoryBean;
21  import org.mybatis.spring.mapper.ClassPathMapperScanner;
22  import org.springframework.beans.BeanUtils;
23  import org.springframework.beans.factory.config.BeanDefinition;
24  import org.springframework.beans.factory.support.BeanNameGenerator;
25  import org.springframework.beans.factory.xml.BeanDefinitionParser;
26  import org.springframework.beans.factory.xml.ParserContext;
27  import org.springframework.beans.factory.xml.XmlReaderContext;
28  import org.springframework.context.ConfigurableApplicationContext;
29  import org.springframework.util.StringUtils;
30  import org.w3c.dom.Element;
31  
32  /**
33   * A {#code BeanDefinitionParser} that handles the element scan of the MyBatis.
34   * namespace
35   * 
36   * @author Lishu Luo
37   * @author Eduardo Macarron
38   *
39   * @since 1.2.0
40   * @see MapperFactoryBean
41   * @see ClassPathMapperScanner
42   * @version $Id$
43   */
44  
45  public class MapperScannerBeanDefinitionParser implements BeanDefinitionParser {
46  
47    private static final String ATTRIBUTE_BASE_PACKAGE = "base-package";
48    private static final String ATTRIBUTE_ANNOTATION = "annotation";
49    private static final String ATTRIBUTE_MARKER_INTERFACE = "marker-interface";
50    private static final String ATTRIBUTE_NAME_GENERATOR = "name-generator";
51    private static final String ATTRIBUTE_TEMPLATE_REF = "template-ref";
52    private static final String ATTRIBUTE_FACTORY_REF = "factory-ref";
53  
54    /**
55     * {@inheritDoc}
56     */
57    @Override
58    public synchronized BeanDefinition parse(Element element, ParserContext parserContext) {
59      ClassPathMapperScanner scanner = new ClassPathMapperScanner(parserContext.getRegistry());
60      ClassLoader classLoader = scanner.getResourceLoader().getClassLoader();
61      XmlReaderContext readerContext = parserContext.getReaderContext();
62      scanner.setResourceLoader(readerContext.getResourceLoader());
63      try {
64        String annotationClassName = element.getAttribute(ATTRIBUTE_ANNOTATION);
65        if (StringUtils.hasText(annotationClassName)) {
66          @SuppressWarnings("unchecked")
67          Class<? extends Annotation> markerInterface = (Class<? extends Annotation>) classLoader.loadClass(annotationClassName);
68          scanner.setAnnotationClass(markerInterface);
69        }
70        String markerInterfaceClassName = element.getAttribute(ATTRIBUTE_MARKER_INTERFACE);
71        if (StringUtils.hasText(markerInterfaceClassName)) {
72          Class<?> markerInterface = classLoader.loadClass(markerInterfaceClassName);
73          scanner.setMarkerInterface(markerInterface);
74        }
75        String nameGeneratorClassName = element.getAttribute(ATTRIBUTE_NAME_GENERATOR);
76        if (StringUtils.hasText(nameGeneratorClassName)) {
77          Class<?> nameGeneratorClass = classLoader.loadClass(nameGeneratorClassName);
78          BeanNameGenerator nameGenerator = BeanUtils.instantiateClass(nameGeneratorClass, BeanNameGenerator.class);
79          scanner.setBeanNameGenerator(nameGenerator);
80        }
81      } catch (Exception ex) {
82        readerContext.error(ex.getMessage(), readerContext.extractSource(element), ex.getCause());
83      }
84      String sqlSessionTemplateBeanName = element.getAttribute(ATTRIBUTE_TEMPLATE_REF);
85      scanner.setSqlSessionTemplateBeanName(sqlSessionTemplateBeanName);
86      String sqlSessionFactoryBeanName = element.getAttribute(ATTRIBUTE_FACTORY_REF);
87      scanner.setSqlSessionFactoryBeanName(sqlSessionFactoryBeanName);
88      scanner.registerFilters();
89      String basePackage = element.getAttribute(ATTRIBUTE_BASE_PACKAGE);
90      scanner.scan(StringUtils.tokenizeToStringArray(basePackage, ConfigurableApplicationContext.CONFIG_LOCATION_DELIMITERS));
91      return null;
92    }
93  
94  }
