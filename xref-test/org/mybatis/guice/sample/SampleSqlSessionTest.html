58  public class SampleSqlSessionTest {

60      private Injector injector;

62      private FooService fooService;

64      @Before
65      public void setupMyBatisGuice() throws Exception {
66  
67          // bindings
68          List<Module> modules = this.createMyBatisModule();
69          this.injector = Guice.createInjector(modules);
70  
71          // prepare the test db
72          Environment environment = this.injector.getInstance(SqlSessionFactory.class).getConfiguration().getEnvironment();
73          DataSource dataSource = environment.getDataSource();
74          ScriptRunner runner = new ScriptRunner(dataSource.getConnection());
75          runner.setAutoCommit(true);
76          runner.setStopOnError(true);
77          runner.runScript(getResourceAsReader("org/mybatis/guice/sample/db/database-schema.sql"));
78          runner.runScript(getResourceAsReader("org/mybatis/guice/sample/db/database-test-data.sql"));
79          runner.closeConnection();
80  
81          this.fooService = this.injector.getInstance(FooService.class);
82      }
83  
84      protected List<Module> createMyBatisModule() {
85          List<Module> modules = new ArrayList<Module>();
86          modules.add(JdbcHelper.HSQLDB_IN_MEMORY_NAMED);
87          modules.add(new MyBatisModule() {
88  
89              @Override
90              protected void initialize() {
91                  bindDataSourceProviderType(PooledDataSourceProvider.class);
92                  bindTransactionFactoryType(JdbcTransactionFactory.class);
93                  addMapperClass(UserMapper.class);
94              }
95  
96          });
97          /* modules.add(new XMLMyBatisModule() {
98  
99              @Override
100             protected void initialize() {
101                 setEnvironmentId("test");
102                 setClassPathResource("org/mybatis/guice/sample/mybatis-config.xml");
103             }
104 
105         }); */
106         modules.add(new Module() {
107             public void configure(Binder binder) {
108                 bindProperties(binder, createTestProperties());
109                 binder.bind(FooService.class).to(FooServiceDaoImpl.class);
110                 binder.bind(UserDao.class).to(UserDaoImpl.class);
111             }
112         });
113 
114         return modules;
115     }
116 
117     protected static Properties createTestProperties() {
118         final Properties myBatisProperties = new Properties();
119         myBatisProperties.setProperty("mybatis.environment.id", "test");
120         myBatisProperties.setProperty("JDBC.username", "sa");
121         myBatisProperties.setProperty("JDBC.password", "");
122         myBatisProperties.setProperty("JDBC.autoCommit", "false");
123         return myBatisProperties;
124     }
