16  package org.mybatis.spring.batch;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  
20  import org.apache.ibatis.session.SqlSession;
21  import org.junit.jupiter.api.Test;
22  import org.mybatis.spring.batch.domain.Employee;
23  import org.springframework.batch.item.Chunk;
24  import org.springframework.beans.factory.annotation.Autowired;
25  import org.springframework.beans.factory.annotation.Qualifier;
26  import org.springframework.test.context.junit.jupiter.SpringJUnitConfig;
27  import org.springframework.transaction.annotation.Transactional;
28  
29  @SpringJUnitConfig(locations = { "classpath:org/mybatis/spring/batch/applicationContext.xml" })
30  class SpringBatchTest {
31  
32    @Autowired
33    @Qualifier("pagingNoNestedItemReader")
34    private MyBatisPagingItemReader<Employee> pagingNoNestedItemReader;
35  
36    @Autowired
37    @Qualifier("pagingNestedItemReader")
38    private MyBatisPagingItemReader<Employee> pagingNestedItemReader;
39  
40    @Autowired
41    @Qualifier("cursorNoNestedItemReader")
42    private MyBatisCursorItemReader<Employee> cursorNoNestedItemReader;
43  
44    @Autowired
45    @Qualifier("cursorNestedItemReader")
46    private MyBatisCursorItemReader<Employee> cursorNestedItemReader;
47  
48    @Autowired
49    private MyBatisBatchItemWriter<Employee> writer;
50  
51    @Autowired
52    private SqlSession session;
53  
54    @Test
55    @Transactional
56    void shouldDuplicateSalaryOfAllEmployees() throws Exception {
57      var employees = new Chunk<Employee>();
58      var employee = pagingNoNestedItemReader.read();
59      while (employee != null) {
60        employee.setSalary(employee.getSalary() * 2);
61        employees.add(employee);
62        employee = pagingNoNestedItemReader.read();
63      }
64      writer.write(employees);
65  
66      assertThat((Integer) session.selectOne("checkSalarySum")).isEqualTo(20000);
67      assertThat((Integer) session.selectOne("checkEmployeeCount")).isEqualTo(employees.size());
68    }
69  
70    @Test
71    @Transactional
72    void checkPagingReadingWithNestedInResultMap() throws Exception {
73      // This test is here to show that PagingReader can return wrong result in case of nested result maps
74      var employees = new Chunk<Employee>();
75      var employee = pagingNestedItemReader.read();
76      while (employee != null) {
77        employee.setSalary(employee.getSalary() * 2);
78        employees.add(employee);
79        employee = pagingNestedItemReader.read();
80      }
81      writer.write(employees);
82  
83      // Assert that we have a WRONG employee count
84      assertThat((Integer) session.selectOne("checkEmployeeCount")).isNotEqualTo(employees.size());
85    }
86  
87    @Test
88    @Transactional
89    void checkCursorReadingWithoutNestedInResultMap() throws Exception {
90      cursorNoNestedItemReader.doOpen();
91      try {
92        var employees = new Chunk<Employee>();
93        var employee = cursorNoNestedItemReader.read();
94        while (employee != null) {
95          employee.setSalary(employee.getSalary() * 2);
96          employees.add(employee);
97          employee = cursorNoNestedItemReader.read();
98        }
99        writer.write(employees);
100 
101       assertThat((Integer) session.selectOne("checkSalarySum")).isEqualTo(20000);
102       assertThat((Integer) session.selectOne("checkEmployeeCount")).isEqualTo(employees.size());
103     } finally {
104       cursorNoNestedItemReader.doClose();
105     }
106   }
107 
108   @Test
109   @Transactional
110   void checkCursorReadingWithNestedInResultMap() throws Exception {
111     cursorNestedItemReader.doOpen();
112     try {
113       var employees = new Chunk<Employee>();
114       var employee = cursorNestedItemReader.read();
115       while (employee != null) {
116         employee.setSalary(employee.getSalary() * 2);
117         employees.add(employee);
118         employee = cursorNestedItemReader.read();
119       }
120       writer.write(employees);
121 
122       assertThat((Integer) session.selectOne("checkSalarySum")).isEqualTo(20000);
123       assertThat((Integer) session.selectOne("checkEmployeeCount")).isEqualTo(employees.size());
124     } finally {
125       cursorNestedItemReader.doClose();
126     }
127   }
128 }
