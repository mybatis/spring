34  @RunWith(SpringJUnit4ClassRunner.class)

36  public class SpringBatchTest {
37  
38    @Autowired
39    private MyBatisPagingItemReader<Employee> reader;
40  
41    @Autowired
42    private MyBatisBatchItemWriter<Employee> writer;
43  
44    @Autowired
45    private SqlSession session;
46  
47    @Test
48    @Transactional
49    public void shouldDuplicateSalaryOfAllEmployees() throws UnexpectedInputException, ParseException, Exception {
50      List<Employee> employees = new ArrayList<Employee>();
51      Employee employee = reader.read();
52      while (employee != null) {
53        employee.setSalary(employee.getSalary() * 2);
54        employees.add(employee);
55        employee = reader.read();
56      }
57      writer.write(employees);
58  
59      assertEquals(20000, session.selectOne("check"));
60    }
