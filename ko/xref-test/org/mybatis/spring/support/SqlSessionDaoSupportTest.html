16  package org.mybatis.spring.support;
17  
18  import static org.assertj.core.api.Assertions.assertThat;
19  import static org.junit.jupiter.api.Assertions.assertThrows;
20  
21  import java.sql.SQLException;
22  
23  import org.junit.jupiter.api.AfterEach;
24  import org.junit.jupiter.api.BeforeEach;
25  import org.junit.jupiter.api.Test;
26  import org.mybatis.spring.AbstractMyBatisSpringTest;
27  import org.mybatis.spring.SqlSessionFactoryBean;
28  import org.mybatis.spring.SqlSessionTemplate;
29  import org.springframework.beans.factory.BeanCreationException;
30  import org.springframework.beans.factory.support.GenericBeanDefinition;
31  import org.springframework.context.annotation.AnnotationConfigUtils;
32  import org.springframework.context.support.GenericApplicationContext;
33  
34  class SqlSessionDaoSupportTest extends AbstractMyBatisSpringTest {
35    private SqlSessionDaoSupport sqlSessionDaoSupport;
36  
37    private GenericApplicationContext applicationContext;
38  
39    @BeforeEach
40    void setup() {
41      sqlSessionDaoSupport = new MockSqlSessionDao();
42    }
43  
44    @AfterEach
45    void closeConnection() throws SQLException {
46      connection.close();
47    }
48  
49    @Test
50    void testWithSqlSessionTemplate() {
51      var sessionTemplate = new SqlSessionTemplate(sqlSessionFactory);
52      sqlSessionDaoSupport.setSqlSessionTemplate(sessionTemplate);
53      sqlSessionDaoSupport.afterPropertiesSet();
54  
55      assertThat(sqlSessionDaoSupport.getSqlSession()).as("should store the Template").isEqualTo(sessionTemplate);
56    }
57  
58    @Test
59    void testWithSqlSessionFactory() {
60      sqlSessionDaoSupport.setSqlSessionFactory(sqlSessionFactory);
61      sqlSessionDaoSupport.afterPropertiesSet();
62  
63      assertThat(((SqlSessionTemplate) sqlSessionDaoSupport.getSqlSession()).getSqlSessionFactory())
64          .as("should store the Factory").isEqualTo(sqlSessionFactory);
65    }
66  
67    @Test
68    void testWithBothFactoryAndTemplate() {
69      var sessionTemplate = new SqlSessionTemplate(sqlSessionFactory);
70      sqlSessionDaoSupport.setSqlSessionTemplate(sessionTemplate);
71      sqlSessionDaoSupport.setSqlSessionFactory(sqlSessionFactory);
72      sqlSessionDaoSupport.afterPropertiesSet();
73  
74      assertThat(sqlSessionDaoSupport.getSqlSession()).as("should ignore the Factory").isEqualTo(sessionTemplate);
75    }
76  
77    @Test
78    void testWithNoFactoryOrSession() {
79      assertThrows(IllegalArgumentException.class, sqlSessionDaoSupport::afterPropertiesSet);
80    }
81  
82    @Test
83    void testAutowireWithNoFactoryOrSession() {
84      setupContext();
85      assertThrows(BeanCreationException.class, this::startContext);
86    }
87  
88    @Test
89    void testAutowireWithTwoFactories() {
90      setupContext();
91  
92      setupSqlSessionFactory("factory1");
93      setupSqlSessionFactory("factory2");
94  
95      assertThrows(BeanCreationException.class, this::startContext);
96    }
97  
98    private void setupContext() {
99      applicationContext = new GenericApplicationContext();
100 
101     var definition = new GenericBeanDefinition();
102     definition.setBeanClass(MockSqlSessionDao.class);
103     applicationContext.registerBeanDefinition("dao", definition);
104 
105     // add support for autowiring fields
106     AnnotationConfigUtils.registerAnnotationConfigProcessors(applicationContext);
107   }
108 
109   private void startContext() {
110     applicationContext.refresh();
111     applicationContext.start();
112 
113     sqlSessionDaoSupport = applicationContext.getBean(MockSqlSessionDao.class);
114   }
115 
116   private void setupSqlSessionFactory(String name) {
117     var definition = new GenericBeanDefinition();
118     definition.setBeanClass(SqlSessionFactoryBean.class);
119     definition.getPropertyValues().add("dataSource", dataSource);
120 
121     applicationContext.registerBeanDefinition(name, definition);
122   }
123 
124   private static final class MockSqlSessionDao extends SqlSessionDaoSupport {
125   }
126 }
