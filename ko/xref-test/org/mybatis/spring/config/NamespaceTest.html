235   @Test
236   void testDefaultScope() {
237     applicationContext = new ClassPathXmlApplicationContext(
238         new String[] { "org/mybatis/spring/config/default-scope.xml" }, false, setupSqlSessionTemplate());
239 
240     startContext();
241 
242     List<String> scopedProxyTargetBeans = Stream.of(applicationContext.getBeanDefinitionNames())
243         .filter(x -> x.startsWith("scopedTarget")).collect(Collectors.toList());
244     assertThat(scopedProxyTargetBeans).hasSize(6).contains("scopedTarget.scopedProxyMapper",
245         "scopedTarget.annotatedMapper", "scopedTarget.annotatedMapperZeroMethods", "scopedTarget.mapperInterface",
246         "scopedTarget.mapperSubinterface", "scopedTarget.mapperChildInterface");
247 
248     for (String scopedProxyTargetBean : scopedProxyTargetBeans) {
249       {
250         BeanDefinition definition = applicationContext.getBeanFactory().getBeanDefinition(scopedProxyTargetBean);
251         assertThat(definition.getBeanClassName()).isEqualTo("org.mybatis.spring.mapper.MapperFactoryBean");
252         assertThat(definition.getScope()).isEqualTo("thread");
253       }
254       {
255         BeanDefinition definition = applicationContext.getBeanFactory()
256             .getBeanDefinition(scopedProxyTargetBean.substring(13));
257         assertThat(definition.getBeanClassName()).isEqualTo("org.springframework.aop.scope.ScopedProxyFactoryBean");
258         assertThat(definition.getScope()).isEqualTo("");
259       }
260     }
261     {
262       ScopedProxyMapper mapper = applicationContext.getBean(ScopedProxyMapper.class);
263       assertThat(mapper.test()).isEqualTo("test");
264     }
265     {
266       AnnotatedMapper mapper = applicationContext.getBean(AnnotatedMapper.class);
267       assertThat(mapper.test()).isEqualTo("main");
268     }
269 
270     SqlSessionFactory sqlSessionFactory = applicationContext.getBean(SqlSessionFactory.class);
271     assertEquals(2, sqlSessionFactory.getConfiguration().getMapperRegistry().getMappers().size());
272   }

305   private GenericApplicationContext setupSqlSessionTemplate() {
306 
307     GenericApplicationContext genericApplicationContext = setupSqlSessionFactory();
308     GenericBeanDefinition definition = new GenericBeanDefinition();
309     definition.setBeanClass(SqlSessionTemplate.class);
310     ConstructorArgumentValues constructorArgs = new ConstructorArgumentValues();
311     constructorArgs.addGenericArgumentValue(new RuntimeBeanReference("sqlSessionFactory"));
312     definition.setConstructorArgumentValues(constructorArgs);
313     genericApplicationContext.registerBeanDefinition("sqlSessionTemplate", definition);
314     return genericApplicationContext;
315   }
316 
317   private GenericApplicationContext setupSqlSessionFactory() {
318 
319     GenericApplicationContext genericApplicationContext = new GenericApplicationContext();
320 
321     GenericBeanDefinition definition = new GenericBeanDefinition();
322     definition.setBeanClass(SqlSessionFactoryBean.class);
323     definition.getPropertyValues().add("dataSource", new MockDataSource());
324 
325     DefaultListableBeanFactory factory = new DefaultListableBeanFactory();
326     factory.registerBeanDefinition("sqlSessionFactory", definition);
327 
328     genericApplicationContext.registerBeanDefinition("sqlSessionFactory", definition);
329 
330     genericApplicationContext.refresh();
331 
332     return genericApplicationContext;
333   }

335   private void assertBeanNotLoaded(String name) {
336     try {
337       applicationContext.getBean(name);
338       fail("Spring bean should not be defined for class " + name);
339     } catch (NoSuchBeanDefinitionException nsbde) {
340       // success
341     }
342   }
