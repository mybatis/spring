16  package org.mybatis.spring.config;
17  
18  import java.lang.annotation.Annotation;
19  
20  import org.mybatis.spring.mapper.MapperFactoryBean;
21  import org.mybatis.spring.mapper.ClassPathMapperScanner;
22  import org.mybatis.spring.mapper.MapperScannerConfigurer;
23  import org.springframework.beans.BeanUtils;
24  import org.springframework.beans.factory.support.AbstractBeanDefinition;
25  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
26  import org.springframework.beans.factory.support.BeanNameGenerator;
27  import org.springframework.beans.factory.xml.AbstractBeanDefinitionParser;
28  import org.springframework.beans.factory.xml.ParserContext;
29  import org.springframework.beans.factory.xml.XmlReaderContext;
30  import org.springframework.util.ClassUtils;
31  import org.springframework.util.StringUtils;
32  import org.w3c.dom.Element;
33  
34  /**
35   * A {#code BeanDefinitionParser} that handles the element scan of the MyBatis. namespace
36   * 
37   * @author Lishu Luo
38   * @author Eduardo Macarron
39   *
40   * @since 1.2.0
41   * @see MapperFactoryBean
42   * @see ClassPathMapperScanner
43   * @see MapperScannerConfigurer
44   */
45  
46  public class MapperScannerBeanDefinitionParser extends AbstractBeanDefinitionParser {
47  
48    private static final String ATTRIBUTE_BASE_PACKAGE = "base-package";
49    private static final String ATTRIBUTE_ANNOTATION = "annotation";
50    private static final String ATTRIBUTE_MARKER_INTERFACE = "marker-interface";
51    private static final String ATTRIBUTE_NAME_GENERATOR = "name-generator";
52    private static final String ATTRIBUTE_TEMPLATE_REF = "template-ref";
53    private static final String ATTRIBUTE_FACTORY_REF = "factory-ref";
54    private static final String ATTRIBUTE_MAPPER_FACTORY_BEAN_CLASS = "mapper-factory-bean-class";
55    private static final String ATTRIBUTE_LAZY_INITIALIZATION = "lazy-initialization";
56  
57    /**
58     * {@inheritDoc}
59     * 
60     * @since 2.0.2
61     */
62    @Override
63    protected AbstractBeanDefinition parseInternal(Element element, ParserContext parserContext) {
64      BeanDefinitionBuilder builder = BeanDefinitionBuilder.genericBeanDefinition(MapperScannerConfigurer.class);
65  
66      ClassLoader classLoader = ClassUtils.getDefaultClassLoader();
67  
68      builder.addPropertyValue("processPropertyPlaceHolders", true);
69      try {
70        String annotationClassName = element.getAttribute(ATTRIBUTE_ANNOTATION);
71        if (StringUtils.hasText(annotationClassName)) {
72          @SuppressWarnings("unchecked")
73          Class<? extends Annotation> annotationClass = (Class<? extends Annotation>) classLoader
74              .loadClass(annotationClassName);
75          builder.addPropertyValue("annotationClass", annotationClass);
76        }
77        String markerInterfaceClassName = element.getAttribute(ATTRIBUTE_MARKER_INTERFACE);
78        if (StringUtils.hasText(markerInterfaceClassName)) {
79          Class<?> markerInterface = classLoader.loadClass(markerInterfaceClassName);
80          builder.addPropertyValue("markerInterface", markerInterface);
81        }
82        String nameGeneratorClassName = element.getAttribute(ATTRIBUTE_NAME_GENERATOR);
83        if (StringUtils.hasText(nameGeneratorClassName)) {
84          Class<?> nameGeneratorClass = classLoader.loadClass(nameGeneratorClassName);
85          BeanNameGenerator nameGenerator = BeanUtils.instantiateClass(nameGeneratorClass, BeanNameGenerator.class);
86          builder.addPropertyValue("nameGenerator", nameGenerator);
87        }
88        String mapperFactoryBeanClassName = element.getAttribute(ATTRIBUTE_MAPPER_FACTORY_BEAN_CLASS);
89        if (StringUtils.hasText(mapperFactoryBeanClassName)) {
90          @SuppressWarnings("unchecked")
91          Class<? extends MapperFactoryBean> mapperFactoryBeanClass = (Class<? extends MapperFactoryBean>) classLoader
92              .loadClass(mapperFactoryBeanClassName);
93          builder.addPropertyValue("mapperFactoryBeanClass", mapperFactoryBeanClass);
94        }
95      } catch (Exception ex) {
96        XmlReaderContext readerContext = parserContext.getReaderContext();
97        readerContext.error(ex.getMessage(), readerContext.extractSource(element), ex.getCause());
98      }
99  
100     builder.addPropertyValue("sqlSessionTemplateBeanName", element.getAttribute(ATTRIBUTE_TEMPLATE_REF));
101     builder.addPropertyValue("sqlSessionFactoryBeanName", element.getAttribute(ATTRIBUTE_FACTORY_REF));
102     builder.addPropertyValue("lazyInitialization", element.getAttribute(ATTRIBUTE_LAZY_INITIALIZATION));
103     builder.addPropertyValue("basePackage", element.getAttribute(ATTRIBUTE_BASE_PACKAGE));
104 
105     return builder.getBeanDefinition();
106   }
107 
108   /**
109    * {@inheritDoc}
110    * 
111    * @since 2.0.2
112    */
113   @Override
114   protected boolean shouldGenerateIdAsFallback() {
115     return true;
116   }
117 
118 }
