266   /**
267    * Set the JDBC {@code DataSource} that this instance should manage transactions for. The {@code DataSource}
268    * should match the one used by the {@code SqlSessionFactory}: for example, you could specify the same
269    * JNDI DataSource for both.
270    *
271    * A transactional JDBC {@code Connection} for this {@code DataSource} will be provided to application code
272    * accessing this {@code DataSource} directly via {@code DataSourceUtils} or {@code DataSourceTransactionManager}.
273    *
274    * The {@code DataSource} specified here should be the target {@code DataSource} to manage transactions for, not
275    * a {@code TransactionAwareDataSourceProxy}. Only data access code may work with
276    * {@code TransactionAwareDataSourceProxy}, while the transaction manager needs to work on the
277    * underlying target {@code DataSource}. If there's nevertheless a {@code TransactionAwareDataSourceProxy}
278    * passed in, it will be unwrapped to extract its target {@code DataSource}.
279    *
280    */
281   public void setDataSource(DataSource dataSource) {
282     if (dataSource instanceof TransactionAwareDataSourceProxy) {
283       // If we got a TransactionAwareDataSourceProxy, we need to perform
284       // transactions for its underlying target DataSource, else data
285       // access code won't see properly exposed transactions (i.e.
286       // transactions for the target DataSource).
287       this.dataSource = ((TransactionAwareDataSourceProxy) dataSource).getTargetDataSource();
288     } else {
289       this.dataSource = dataSource;
290     }
291   }

343   /**
344    * Build a {@code SqlSessionFactory} instance.
345    *
346    * The default implementation uses the standard MyBatis {@code XMLConfigBuilder} API to build a
347    * {@code SqlSessionFactory} instance based on an Reader.
348    *
349    * @return SqlSessionFactory
350    * @throws IOException if loading the config file failed
351    */
352   protected SqlSessionFactory buildSqlSessionFactory() throws IOException {
353 
354     Configuration configuration;
355 
356     XMLConfigBuilder xmlConfigBuilder = null;
357     if (this.configLocation != null) {
358       xmlConfigBuilder = new XMLConfigBuilder(this.configLocation.getInputStream(), null, this.configurationProperties);
359       configuration = xmlConfigBuilder.getConfiguration();
360     } else {
361       if (logger.isDebugEnabled()) {
362         logger.debug("Property 'configLocation' not specified, using default MyBatis Configuration");
363       }
364       configuration = new Configuration();
365       configuration.setVariables(this.configurationProperties);
366     }
367 
368     if (this.objectFactory != null) {
369       configuration.setObjectFactory(this.objectFactory);
370     }
371 
372     if (this.objectWrapperFactory != null) {
373       configuration.setObjectWrapperFactory(this.objectWrapperFactory);
374     }
375 
376     if (hasLength(this.typeAliasesPackage)) {
377       String[] typeAliasPackageArray = tokenizeToStringArray(this.typeAliasesPackage,
378           ConfigurableApplicationContext.CONFIG_LOCATION_DELIMITERS);
379       for (String packageToScan : typeAliasPackageArray) {
380         configuration.getTypeAliasRegistry().registerAliases(packageToScan,
381                 typeAliasesSuperType == null ? Object.class : typeAliasesSuperType);
382         if (logger.isDebugEnabled()) {
383           logger.debug("Scanned package: '" + packageToScan + "' for aliases");
384         }
385       }
386     }
387 
388     if (!isEmpty(this.typeAliases)) {
389       for (Class<?> typeAlias : this.typeAliases) {
390         configuration.getTypeAliasRegistry().registerAlias(typeAlias);
391         if (logger.isDebugEnabled()) {
392           logger.debug("Registered type alias: '" + typeAlias + "'");
393         }
394       }
395     }
396 
397     if (!isEmpty(this.plugins)) {
398       for (Interceptor plugin : this.plugins) {
399         configuration.addInterceptor(plugin);
400         if (logger.isDebugEnabled()) {
401           logger.debug("Registered plugin: '" + plugin + "'");
402         }
403       }
404     }
405 
406     if (hasLength(this.typeHandlersPackage)) {
407       String[] typeHandlersPackageArray = tokenizeToStringArray(this.typeHandlersPackage,
408           ConfigurableApplicationContext.CONFIG_LOCATION_DELIMITERS);
409       for (String packageToScan : typeHandlersPackageArray) {
410         configuration.getTypeHandlerRegistry().register(packageToScan);
411         if (logger.isDebugEnabled()) {
412           logger.debug("Scanned package: '" + packageToScan + "' for type handlers");
413         }
414       }
415     }
416 
417     if (!isEmpty(this.typeHandlers)) {
418       for (TypeHandler<?> typeHandler : this.typeHandlers) {
419         configuration.getTypeHandlerRegistry().register(typeHandler);
420         if (logger.isDebugEnabled()) {
421           logger.debug("Registered type handler: '" + typeHandler + "'");
422         }
423       }
424     }
425 
426     if (xmlConfigBuilder != null) {
427       try {
428         xmlConfigBuilder.parse();
429 
430         if (logger.isDebugEnabled()) {
431           logger.debug("Parsed configuration file: '" + this.configLocation + "'");
432         }
433       } catch (Exception ex) {
434         throw new NestedIOException("Failed to parse config resource: " + this.configLocation, ex);
435       } finally {
436         ErrorContext.instance().reset();
437       }
438     }
439 
440     if (this.transactionFactory == null) {
441       this.transactionFactory = new SpringManagedTransactionFactory();
442     }
443 
444     Environment environment = new Environment(this.environment, this.transactionFactory, this.dataSource);
445     configuration.setEnvironment(environment);
446 
447     if (this.databaseIdProvider != null) {
448       try {
449         configuration.setDatabaseId(this.databaseIdProvider.getDatabaseId(this.dataSource));
450       } catch (SQLException e) {
451         throw new NestedIOException("Failed getting a databaseId", e);
452       }
453     }
454 
455     if (!isEmpty(this.mapperLocations)) {
456       for (Resource mapperLocation : this.mapperLocations) {
457         if (mapperLocation == null) {
458           continue;
459         }
460 
461         try {
462           XMLMapperBuilder xmlMapperBuilder = new XMLMapperBuilder(mapperLocation.getInputStream(),
463               configuration, mapperLocation.toString(), configuration.getSqlFragments());
464           xmlMapperBuilder.parse();
465         } catch (Exception e) {
466           throw new NestedIOException("Failed to parse mapping resource: '" + mapperLocation + "'", e);
467         } finally {
468           ErrorContext.instance().reset();
469         }
470 
471         if (logger.isDebugEnabled()) {
472           logger.debug("Parsed mapper file: '" + mapperLocation + "'");
473         }
474       }
475     } else {
476       if (logger.isDebugEnabled()) {
477         logger.debug("Property 'mapperLocations' was not specified or no matching resources found");
478       }
479     }
480 
481     return this.sqlSessionFactoryBuilder.build(configuration);
482   }
