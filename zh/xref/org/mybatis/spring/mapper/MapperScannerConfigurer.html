44  /**
45   * BeanDefinitionRegistryPostProcessor that searches recursively starting from a base package for
46   * interfaces and registers them as {@code MapperFactoryBean}. Note that only interfaces with at
47   * least one method will be registered; concrete classes will be ignored.
48   * <p>
49   * This class was a {code BeanFactoryPostProcessor} until 1.0.1 version. It changed to  
50   * {@code BeanDefinitionRegistryPostProcessor} in 1.0.2. See https://jira.springsource.org/browse/SPR-8269
51   * for the details.
52   * <p>
53   * The {@code basePackage} property can contain more than one package name, separated by either
54   * commas or semicolons.
55   * <p>
56   * This class supports filtering the mappers created by either specifying a marker interface or an
57   * annotation. The {@code annotationClass} property specifies an annotation to search for. The
58   * {@code markerInterface} property specifies a parent interface to search for. If both properties
59   * are specified, mappers are added for interfaces that match <em>either</em> criteria. By default,
60   * these two properties are null, so all interfaces in the given {@code basePackage} are added as
61   * mappers.
62   * <p>
63   * This configurer enables autowire for all the beans that it creates so that they are
64   * automatically autowired with the proper {@code SqlSessionFactory} or {@code SqlSessionTemplate}.
65   * If there is more than one {@code SqlSessionFactory} in the application, however, autowiring
66   * cannot be used. In this case you must explicitly specify either an {@code SqlSessionFactory} or
67   * an {@code SqlSessionTemplate} to use via the <em>bean name</em> properties. Bean names are used
68   * rather than actual objects because Spring does not initialize property placeholders until after
69   * this class is processed. 
70   * <p>
71   * Passing in an actual object which may require placeholders (i.e. DB user password) will fail. 
72   * Using bean names defers actual object creation until later in the startup
73   * process, after all placeholder substituation is completed. However, note that this configurer
74   * does support property placeholders of its <em>own</em> properties. The <code>basePackage</code>
75   * and bean name properties all support <code>${property}</code> style substitution.
76   * <p>
77   * Configuration sample:
78   * <p>
79   *
80   * <pre class="code">
81   * {@code
82   *   <bean class="org.mybatis.spring.mapper.MapperScannerConfigurer">
83   *       <property name="basePackage" value="org.mybatis.spring.sample.mapper" />
84   *       <!-- optional unless there are multiple session factories defined -->
85   *       <property name="sqlSessionFactoryBeanName" value="sqlSessionFactory" />
86   *   </bean>
87   * }
88   * </pre>
89   *
90   * @author Hunter Presnall
91   * @author Eduardo Macarron
92   *
93   * @see MapperFactoryBean
94   * @see ClassPathMapperScanner
95   * @version $Id$
96   */

294   /**
295    * {@inheritDoc}
296    * 
297    * @since 1.0.2
298    */
299   public void postProcessBeanDefinitionRegistry(BeanDefinitionRegistry registry) throws BeansException {
300     if (this.processPropertyPlaceHolders) {
301       processPropertyPlaceHolders();
302     }
303 
304     ClassPathMapperScanner scanner = new ClassPathMapperScanner(registry);
305     scanner.setAddToConfig(this.addToConfig);
306     scanner.setAnnotationClass(this.annotationClass);
307     scanner.setMarkerInterface(this.markerInterface);
308     scanner.setSqlSessionFactory(this.sqlSessionFactory);
309     scanner.setSqlSessionTemplate(this.sqlSessionTemplate);
310     scanner.setSqlSessionFactoryBeanName(this.sqlSessionFactoryBeanName);
311     scanner.setSqlSessionTemplateBeanName(this.sqlSessionTemplateBeanName);
312     scanner.setResourceLoader(this.applicationContext);
313     scanner.setBeanNameGenerator(this.nameGenerator);
314     scanner.registerFilters();
315     scanner.scan(StringUtils.tokenizeToStringArray(this.basePackage, ConfigurableApplicationContext.CONFIG_LOCATION_DELIMITERS));
316   }

318   /*
319    * BeanDefinitionRegistries are called early in application startup, before
320    * BeanFactoryPostProcessors. This means that PropertyResourceConfigurers will not have been
321    * loaded and any property substitution of this class' properties will fail. To avoid this, find
322    * any PropertyResourceConfigurers defined in the context and run them on this class' bean
323    * definition. Then update the values.
324    */
325   private void processPropertyPlaceHolders() {
326     Map<String, PropertyResourceConfigurer> prcs = applicationContext.getBeansOfType(PropertyResourceConfigurer.class);
327 
328     if (!prcs.isEmpty() && applicationContext instanceof GenericApplicationContext) {
329       BeanDefinition mapperScannerBean = ((GenericApplicationContext) applicationContext)
330           .getBeanFactory().getBeanDefinition(beanName);
331 
332       // PropertyResourceConfigurer does not expose any methods to explicitly perform
333       // property placeholder substitution. Instead, create a BeanFactory that just
334       // contains this mapper scanner and post process the factory.
335       DefaultListableBeanFactory factory = new DefaultListableBeanFactory();
336       factory.registerBeanDefinition(beanName, mapperScannerBean);
337 
338       for (PropertyResourceConfigurer prc : prcs.values()) {
339         prc.postProcessBeanFactory(factory);
340       }
341 
342       PropertyValues values = mapperScannerBean.getPropertyValues();
343 
344       this.basePackage = updatePropertyValue("basePackage", values);
345       this.sqlSessionFactoryBeanName = updatePropertyValue("sqlSessionFactoryBeanName", values);
346       this.sqlSessionTemplateBeanName = updatePropertyValue("sqlSessionTemplateBeanName", values);
347     }
348   }
349 
350   private String updatePropertyValue(String propertyName, PropertyValues values) {
351     PropertyValue property = values.getPropertyValue(propertyName);
352 
353     if (property == null) {
354       return null;
355     }
356 
357     Object value = property.getValue();
358 
359     if (value == null) {
360       return null;
361     } else if (value instanceof String) {
362       return value.toString();
363     } else if (value instanceof TypedStringValue) {
364       return ((TypedStringValue) value).getValue();
365     } else {
366       return null;
367     }
368   }
