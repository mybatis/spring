79      /**
80       * {@inheritDoc}
81       */
82      @Override
83      final void internalConfigure() {
84          checkState( aliases == null, "Re-entry is not allowed." );
85          checkState( handlers == null, "Re-entry is not allowed." );
86          checkState( interceptors == null, "Re-entry is not allowed." );
87          checkState( mappers == null, "Re-entry is not allowed." );
88  
89          aliases = newMapBinder(binder(), new TypeLiteral<String>(){}, new TypeLiteral<Class<?>>(){}, TypeAliases.class);
90          handlers = newMapBinder(binder(), new TypeLiteral<Class<?>>(){}, new TypeLiteral<TypeHandler<?>>(){});
91          interceptors = newSetBinder(binder(), Interceptor.class);
92          mappingTypeHandlers = newSetBinder(binder(), new TypeLiteral<TypeHandler<?>>(){}, MappingTypeHandlers.class);
93          mappers = newSetBinder(binder(), new TypeLiteral<Class<?>>(){}, Mappers.class);
94  
95          try {
96              initialize();
97          } finally {
98              aliases = null;
99              handlers = null;
100             interceptors = null;
101             mappers = null;
102         }
103 
104         // fixed bindings
105         bind(Environment.class).toProvider(EnvironmentProvider.class).in(Scopes.SINGLETON);
106         bind(Configuration.class).toProvider(ConfigurationProvider.class).in(Scopes.SINGLETON);
107         bind(SqlSessionFactory.class).toProvider(SqlSessionFactoryProvider.class).in(Scopes.SINGLETON);
108 
109         // parametric bindings
110         bind(ObjectFactory.class).to(objectFactoryType).in(Scopes.SINGLETON);
111     }

278     /**
279      * Add a user defined binding.
280      *
281      * @param alias the string type alias
282      * @param clazz the type has to be bound.
283      */
284     protected final AliasBinder addAlias(final String alias) {
285         checkArgument(alias != null && alias.length() > 0, "Empty or null 'alias' is not valid");
286 
287         return new AliasBinder() {
288 
289             public void to(final Class<?> clazz) {
290                 checkArgument(clazz != null, "Null type not valid for alias '%s'", alias);
291                 aliases.addBinding(alias).toInstance(clazz);
292             }
293 
294         };
295     }
