40  /**
41   * Handles MyBatis SqlSession life cycle. It can register and get SqlSessions from
42   * Spring {@code TransactionSynchronizationManager}. Also works if no transaction is active.
43   *
44   * @author Hunter Presnall 
45   * @author Eduardo Macarron
46   *
47   * @version $Id$
48   */
49  public final class SqlSessionUtils {
50  
51    private static final Log logger = LogFactory.getLog(SqlSessionUtils.class);
52  
53    /**
54     * This class can't be instantiated, exposes static utility methods only.
55     */
56    private SqlSessionUtils() {
57      // do nothing
58    }

60    /**
61     * Creates a new MyBatis {@code SqlSession} from the {@code SqlSessionFactory}
62     * provided as a parameter and using its {@code DataSource} and {@code ExecutorType}
63     *
64     * @param sessionFactory a MyBatis {@code SqlSessionFactory} to create new sessions
65     * @return a MyBatis {@code SqlSession}
66     * @throws TransientDataAccessResourceException if a transaction is active and the
67     *             {@code SqlSessionFactory} is not using a {@code SpringManagedTransactionFactory}
68     */
69    public static SqlSession getSqlSession(SqlSessionFactory sessionFactory) {
70      ExecutorType executorType = sessionFactory.getConfiguration().getDefaultExecutorType();
71      return getSqlSession(sessionFactory, executorType, null);
72    }

74    /**
75     * Gets an SqlSession from Spring Transaction Manager or creates a new one if needed.
76     * Tries to get a SqlSession out of current transaction. If there is not any, it creates a new one.
77     * Then, it synchronizes the SqlSession with the transaction if Spring TX is active and
78     * <code>SpringManagedTransactionFactory</code> is configured as a transaction manager.
79     *
80     * @param sessionFactory a MyBatis {@code SqlSessionFactory} to create new sessions
81     * @param executorType The executor type of the SqlSession to create
82     * @param exceptionTranslator Optional. Translates SqlSession.commit() exceptions to Spring exceptions.
83     * @throws TransientDataAccessResourceException if a transaction is active and the
84     *             {@code SqlSessionFactory} is not using a {@code SpringManagedTransactionFactory}
85     * @see SpringManagedTransactionFactory
86     */
87    public static SqlSession getSqlSession(SqlSessionFactory sessionFactory, ExecutorType executorType, PersistenceExceptionTranslator exceptionTranslator) {
88  
89      notNull(sessionFactory, "No SqlSessionFactory specified");
90      notNull(executorType, "No ExecutorType specified");
91  
92      SqlSessionHolder holder = (SqlSessionHolder) getResource(sessionFactory);
93  
94      if (holder != null && holder.isSynchronizedWithTransaction()) {
95        if (holder.getExecutorType() != executorType) {
96          throw new TransientDataAccessResourceException("Cannot change the ExecutorType when there is an existing transaction");
97        }
98  
99        holder.requested();
100 
101       if (logger.isDebugEnabled()) {
102         logger.debug("Fetched SqlSession [" + holder.getSqlSession() + "] from current transaction");
103       }
104 
105       return holder.getSqlSession();
106     }
107 
108     if (logger.isDebugEnabled()) {
109       logger.debug("Creating a new SqlSession");
110     }
111 
112     SqlSession session = sessionFactory.openSession(executorType);
113 
114     // Register session holder if synchronization is active (i.e. a Spring TX is active)
115     //
116     // Note: The DataSource used by the Environment should be synchronized with the
117     // transaction either through DataSourceTxMgr or another tx synchronization.
118     // Further assume that if an exception is thrown, whatever started the transaction will
119     // handle closing / rolling back the Connection associated with the SqlSession.
120     if (isSynchronizationActive()) {
121       Environment environment = sessionFactory.getConfiguration().getEnvironment();
122 
123       if (environment.getTransactionFactory() instanceof SpringManagedTransactionFactory) {
124         if (logger.isDebugEnabled()) {
125           logger.debug("Registering transaction synchronization for SqlSession [" + session + "]");
126         }
127 
128         holder = new SqlSessionHolder(session, executorType, exceptionTranslator);
129         bindResource(sessionFactory, holder);
130         registerSynchronization(new SqlSessionSynchronization(holder, sessionFactory));
131         holder.setSynchronizedWithTransaction(true);
132         holder.requested();
133       } else {
134         if (getResource(environment.getDataSource()) == null) {
135           if (logger.isDebugEnabled()) {
136             logger.debug("SqlSession [" + session + "] was not registered for synchronization because DataSource is not transactional");
137           }
138         } else {
139           throw new TransientDataAccessResourceException(
140               "SqlSessionFactory must be using a SpringManagedTransactionFactory in order to use Spring transaction synchronization");
141         }
142       }
143     } else {
144       if (logger.isDebugEnabled()) {
145         logger.debug("SqlSession [" + session + "] was not registered for synchronization because synchronization is not active");
146       }
147     }
148 
149     return session;
150   }

152   /**
153    * Checks if {@code SqlSession} passed as an argument is managed by Spring {@code TransactionSynchronizationManager}
154    * If it is not, it closes it, otherwise it just updates the reference counter and
155    * lets Spring call the close callback when the managed transaction ends
156    *
157    * @param session
158    * @param sessionFactory
159    */
160   public static void closeSqlSession(SqlSession session, SqlSessionFactory sessionFactory) {
161 
162     notNull(session, "No SqlSession specified");
163     notNull(sessionFactory, "No SqlSessionFactory specified");
164 
165     SqlSessionHolder holder = (SqlSessionHolder) getResource(sessionFactory);
166     if ((holder != null) && (holder.getSqlSession() == session)) {
167       if (logger.isDebugEnabled()) {
168         logger.debug("Releasing transactional SqlSession [" + session + "]");
169       }
170       holder.released();
171     } else {
172       if (logger.isDebugEnabled()) {
173         logger.debug("Closing non transactional SqlSession [" + session + "]");
174       }
175       session.close();
176     }
177   }

179   /**
180    * Returns if the {@code SqlSession} passed as an argument is being managed by Spring
181    *
182    * @param session a MyBatis SqlSession to check
183    * @param sessionFactory the SqlSessionFactory which the SqlSession was built with
184    * @return true if session is transactional, otherwise false
185    */
186   public static boolean isSqlSessionTransactional(SqlSession session, SqlSessionFactory sessionFactory) {
187     notNull(session, "No SqlSession specified");
188     notNull(sessionFactory, "No SqlSessionFactory specified");
189 
190     SqlSessionHolder holder = (SqlSessionHolder) getResource(sessionFactory);
191 
192     return (holder != null) && (holder.getSqlSession() == session);
193   }

195   /**
196    * Callback for cleaning up resources. It cleans TransactionSynchronizationManager and
197    * also commits and closes the {@code SqlSession}.
198    * It assumes that {@code Connection} life cycle will be managed by
199    * {@code DataSourceTransactionManager} or {@code JtaTransactionManager}
200    */
201   private static final class SqlSessionSynchronization extends TransactionSynchronizationAdapter {
202 
203     private final SqlSessionHolder holder;
204 
205     private final SqlSessionFactory sessionFactory;
206 
207     public SqlSessionSynchronization(SqlSessionHolder holder, SqlSessionFactory sessionFactory) {
208       notNull(holder, "Parameter 'holder' must be not null");
209       notNull(sessionFactory, "Parameter 'sessionFactory' must be not null");
210 
211       this.holder = holder;
212       this.sessionFactory = sessionFactory;
213     }
214 
215     /**
216      * {@inheritDoc}
217      */
218     @Override
219     public int getOrder() {
220       // order right before any Connection synchronization
221       return DataSourceUtils.CONNECTION_SYNCHRONIZATION_ORDER - 1;
222     }
223 
224     /**
225      * {@inheritDoc}
226      */
227     @Override
228     public void suspend() {
229       unbindResource(this.sessionFactory);
230     }
231 
232     /**
233      * {@inheritDoc}
234      */
235     @Override
236     public void resume() {
237       bindResource(this.sessionFactory, this.holder);
238     }
239 
240     /**
241      * {@inheritDoc}
242      */
243     @Override
244     public void beforeCommit(boolean readOnly) {
245       // Flush BATCH statements so they are actually executed before the connection is committed.
246       // If there is no tx active data will be rolled back so there is no need to flush batches
247       if (this.holder.getExecutorType() == ExecutorType.BATCH && isActualTransactionActive()) {
248         try {
249           if (logger.isDebugEnabled()) {
250             logger.debug("Transaction synchronization flushing SqlSession [" + this.holder.getSqlSession() + "]");
251           }
252           this.holder.getSqlSession().flushStatements();
253         } catch (PersistenceException p) {
254           if (this.holder.getPersistenceExceptionTranslator() != null) {
255             DataAccessException translated = this.holder.getPersistenceExceptionTranslator().translateExceptionIfPossible(p);
256             if (translated != null) {
257               throw translated;
258             }
259           }
260           throw p;
261         }
262       }
263     }
264 
265     /**
266      * {@inheritDoc}
267      */
268     @Override
269     public void afterCompletion(int status) {
270       // Unbind the SqlSession from tx synchronization
271       // Note, commit/rollback is needed to ensure 2nd level cache is properly updated
272       // SpringTransaction will no-op the connection commit/rollback
273       try {
274         // Do not call commit unless there is really a transaction; 
275         // no need to commit if just tx synchronization is active but no transaction was started
276         if (isActualTransactionActive()) {
277           switch (status) {
278           case STATUS_COMMITTED:
279             if (logger.isDebugEnabled()) {
280               logger.debug("Transaction synchronization committing SqlSession [" + this.holder.getSqlSession() + "]");
281             }
282             holder.getSqlSession().commit();
283             break;
284           case STATUS_ROLLED_BACK:
285             if (logger.isDebugEnabled()) {
286               logger.debug("Transaction synchronization rolling back SqlSession [" + this.holder.getSqlSession() + "]");
287             }
288             holder.getSqlSession().rollback();
289             break;
290           default:
291             if (logger.isDebugEnabled()) {
292               logger.debug("Transaction synchronization ended with unknown status for SqlSession [" + this.holder.getSqlSession() + "]");
293             }
294           }
295         }
296       } finally {
297         if (!holder.isOpen()) {
298           unbindResource(sessionFactory);
299           try {
300             if (logger.isDebugEnabled()) {
301               logger.debug("Transaction synchronization closing SqlSession [" + this.holder.getSqlSession() + "]");
302             }
303             this.holder.getSqlSession().close();
304           } finally {
305             this.holder.reset();
306           }
307         }
308       }
309     }
310   }
