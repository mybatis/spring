16  package org.mybatis.spring;

18  import static org.assertj.core.api.Assertions.assertThat;
19  import static org.junit.jupiter.api.Assertions.fail;
20  
21  import java.sql.SQLException;
22  
23  import org.apache.ibatis.plugin.Interceptor;
24  import org.apache.ibatis.session.SqlSessionFactory;
25  import org.junit.jupiter.api.AfterEach;
26  import org.junit.jupiter.api.BeforeEach;
27  import org.junit.jupiter.api.BeforeAll;
28  import org.springframework.core.io.ClassPathResource;
29  import org.springframework.core.io.Resource;
30  import org.springframework.dao.support.PersistenceExceptionTranslator;
31  import org.springframework.jdbc.datasource.DataSourceTransactionManager;
32  
33  import com.mockrunner.mock.jdbc.MockConnection;
34  import com.mockrunner.mock.jdbc.MockResultSet;
35  
36  public abstract class AbstractMyBatisSpringTest {
37  
38    protected static PooledMockDataSourceedMockDataSource">PooledMockDataSource dataSource = new PooledMockDataSource();
39  
40    protected static SqlSessionFactory sqlSessionFactory;
41  
42    protected static ExecutorInterceptor executorInterceptor = new ExecutorInterceptor();
43  
44    protected static DataSourceTransactionManager txManager;
45  
46    protected static PersistenceExceptionTranslator exceptionTranslator;
47  
48    protected MockConnection connection;
49  
50    protected MockConnection connectionTwo;
51  
52    @BeforeAll
53    public static void setupBase() throws Exception {
54      // create an SqlSessionFactory that will use SpringManagedTransactions
55      SqlSessionFactoryBean factoryBean = new SqlSessionFactoryBean();
56      factoryBean.setMapperLocations(new Resource[] { new ClassPathResource("org/mybatis/spring/TestMapper.xml") });
57      // note running without SqlSessionFactoryBean.configLocation set => default configuration
58      factoryBean.setDataSource(dataSource);
59      factoryBean.setPlugins(new Interceptor[] { executorInterceptor });
60  
61      exceptionTranslator = new MyBatisExceptionTranslator(dataSource, true);
62  
63      sqlSessionFactory = factoryBean.getObject();
64  
65      txManager = new DataSourceTransactionManager(dataSource);
66    }
67  
68    protected void assertNoCommit() {
69      assertNoCommitJdbc();
70      assertNoCommitSession();
71    }
72  
73    protected void assertNoCommitJdbc() {
74      assertThat(connection.getNumberCommits()).as("should not call commit on Connection").isEqualTo(0);
75      assertThat(connection.getNumberRollbacks()).as("should not call rollback on Connection").isEqualTo(0);
76    }
77  
78    protected void assertNoCommitSession() {
79      assertThat(executorInterceptor.getCommitCount()).as("should not call commit on SqlSession").isEqualTo(0);
80      assertThat(executorInterceptor.getRollbackCount()).as("should not call rollback on SqlSession").isEqualTo(0);
81    }
82  
83    protected void assertCommit() {
84      assertCommitJdbc();
85      assertCommitSession();
86    }
87  
88    protected void assertCommitJdbc() {
89      assertThat(connection.getNumberCommits()).as("should call commit on Connection").isEqualTo(1);
90      assertThat(connection.getNumberRollbacks()).as("should not call rollback on Connection").isEqualTo(0);
91    }
92  
93    protected void assertCommitSession() {
94      assertThat(executorInterceptor.getCommitCount()).as("should call commit on SqlSession").isEqualTo(1);
95      assertThat(executorInterceptor.getRollbackCount()).as("should not call rollback on SqlSession").isEqualTo(0);
96    }
97  
98    protected void assertRollback() {
99      assertThat(connection.getNumberCommits()).as("should not call commit on Connection").isEqualTo(0);
100     assertThat(connection.getNumberRollbacks()).as("should call rollback on Connection").isEqualTo(1);
101     assertThat(executorInterceptor.getCommitCount()).as("should not call commit on SqlSession").isEqualTo(0);
102     assertThat(executorInterceptor.getRollbackCount()).as("should call rollback on SqlSession").isEqualTo(1);
103   }
104 
105   protected void assertSingleConnection() {
106     assertThat(dataSource.getConnectionCount()).as("should only call DataSource.getConnection() once").isEqualTo(1);
107   }
108 
109   protected void assertExecuteCount(int count) {
110     assertThat(connection.getPreparedStatementResultSetHandler().getExecutedStatements().size()).as(
111         "should have executed %d SQL statements", count).isEqualTo(count);
112   }
113 
114   protected void assertConnectionClosed(MockConnection connection) {
115     try {
116       if ((connection != null) && !connection.isClosed()) {
117         fail("Connection is not closed");
118       }
119     } catch (SQLException sqle) {
120       fail("cannot call Connection.isClosed() " + sqle.getMessage());
121     }
122   }
123 
124   protected MockConnection createMockConnection() {
125     // this query must be the same as the query in TestMapper.xml
126     MockResultSet rs = new MockResultSet("SELECT 1");
127     rs.addRow(new Object[] { 1 });
128 
129     MockConnection con = new MockConnection();
130     con.getPreparedStatementResultSetHandler().prepareResultSet("SELECT 1", rs);
131 
132     return con;
133   }
134 
135   /*
136    * Setup a new Connection before each test since its closed state will need to be checked
137    * afterwards and there is no Connection.open().
138    */
139   @BeforeEach
140   public void setupConnection() throws SQLException {
141     dataSource.reset();
142     connection = createMockConnection();
143     connectionTwo = createMockConnection();
144     dataSource.addConnection(connectionTwo);
145     dataSource.addConnection(connection);
146   }
147 
148   @BeforeEach
149   public void resetExecutorInterceptor() {
150     executorInterceptor.reset();
151   }
152 
153   @AfterEach
154   public void validateConnectionClosed() {
155     assertConnectionClosed(connection);
156 
157     connection = null;
158   }
