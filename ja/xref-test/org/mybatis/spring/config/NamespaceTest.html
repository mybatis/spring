235   @Test
236   void testDefaultScope() {
237     applicationContext = new ClassPathXmlApplicationContext(
238         new String[] { "org/mybatis/spring/config/default-scope.xml" }, false, setupSqlSessionTemplate());
239 
240     startContext();
241 
242     List<String> scopedProxyTargetBeans = Stream.of(applicationContext.getBeanDefinitionNames())
243         .filter(x -> x.startsWith("scopedTarget")).collect(Collectors.toList());
244     assertThat(scopedProxyTargetBeans).hasSize(6).contains("scopedTarget.scopedProxyMapper",
245         "scopedTarget.annotatedMapper", "scopedTarget.annotatedMapperZeroMethods", "scopedTarget.mapperInterface",
246         "scopedTarget.mapperSubinterface", "scopedTarget.mapperChildInterface");
247 
248     for (String scopedProxyTargetBean : scopedProxyTargetBeans) {
249       {
250         var definition = applicationContext.getBeanFactory().getBeanDefinition(scopedProxyTargetBean);
251         assertThat(definition.getBeanClassName()).isEqualTo("org.mybatis.spring.mapper.MapperFactoryBean");
252         assertThat(definition.getScope()).isEqualTo("thread");
253       }
254       {
255         var definition = applicationContext.getBeanFactory().getBeanDefinition(scopedProxyTargetBean.substring(13));
256         assertThat(definition.getBeanClassName()).isEqualTo("org.springframework.aop.scope.ScopedProxyFactoryBean");
257         assertThat(definition.getScope()).isEqualTo("");
258       }
259     }
260     {
261       var mapper = applicationContext.getBean(ScopedProxyMapper.class);
262       assertThat(mapper.test()).isEqualTo("test");
263     }
264     {
265       var mapper = applicationContext.getBean(AnnotatedMapper.class);
266       assertThat(mapper.test()).isEqualTo("main");
267     }
268 
269     var sqlSessionFactory = applicationContext.getBean(SqlSessionFactory.class);
270     assertEquals(2, sqlSessionFactory.getConfiguration().getMapperRegistry().getMappers().size());
271   }

304   private GenericApplicationContext setupSqlSessionTemplate() {
305 
306     var genericApplicationContext = setupSqlSessionFactory();
307     var definition = new GenericBeanDefinition();
308     definition.setBeanClass(SqlSessionTemplate.class);
309     var constructorArgs = new ConstructorArgumentValues();
310     constructorArgs.addGenericArgumentValue(new RuntimeBeanReference("sqlSessionFactory"));
311     definition.setConstructorArgumentValues(constructorArgs);
312     genericApplicationContext.registerBeanDefinition("sqlSessionTemplate", definition);
313     return genericApplicationContext;
314   }
315 
316   private GenericApplicationContext setupSqlSessionFactory() {
317 
318     var genericApplicationContext = new GenericApplicationContext();
319 
320     var definition = new GenericBeanDefinition();
321     definition.setBeanClass(SqlSessionFactoryBean.class);
322     definition.getPropertyValues().add("dataSource", new MockDataSource());
323 
324     var factory = new DefaultListableBeanFactory();
325     factory.registerBeanDefinition("sqlSessionFactory", definition);
326 
327     genericApplicationContext.registerBeanDefinition("sqlSessionFactory", definition);
328 
329     genericApplicationContext.refresh();
330 
331     return genericApplicationContext;
332   }

334   private void assertBeanNotLoaded(String name) {
335     try {
336       applicationContext.getBean(name);
337       fail("Spring bean should not be defined for class " + name);
338     } catch (NoSuchBeanDefinitionException nsbde) {
339       // success
340     }
341   }
