27    @Test
28    void shouldNoOpWithTx() throws Exception {
29      DefaultTransactionDefinition txDef = new DefaultTransactionDefinition();
30      txDef.setPropagationBehaviorName("PROPAGATION_REQUIRED");
31      TransactionStatus status = txManager.getTransaction(txDef);
32  
33      SpringManagedTransactionFactory transactionFactory = new SpringManagedTransactionFactory();
34      SpringManagedTransaction transaction = (SpringManagedTransaction) transactionFactory.newTransaction(dataSource,
35          null, false);
36      transaction.getConnection();
37      transaction.commit();
38      transaction.close();
39      assertThat(connection.getNumberCommits()).as("should not call commit on Connection").isEqualTo(0);
40      assertThat(connection.isClosed()).as("should not close the Connection").isFalse();
41  
42      txManager.commit(status);
43    }

62    @Test
63    void shouldManageWithNoTx() throws Exception {
64      SpringManagedTransactionFactory transactionFactory = new SpringManagedTransactionFactory();
65      SpringManagedTransaction transaction = (SpringManagedTransaction) transactionFactory.newTransaction(dataSource,
66          null, false);
67      transaction.getConnection();
68      transaction.commit();
69      transaction.close();
70      assertThat(connection.getNumberCommits()).as("should call commit on Connection").isEqualTo(1);
71      assertThat(connection.isClosed()).as("should close the Connection").isTrue();
72    }
73  
74    @Test
75    void shouldNotCommitWithNoTxAndAutocommitIsOn() throws Exception {
76      SpringManagedTransactionFactory transactionFactory = new SpringManagedTransactionFactory();
77      SpringManagedTransaction transaction = (SpringManagedTransaction) transactionFactory.newTransaction(dataSource,
78          null, false);
79      connection.setAutoCommit(true);
80      transaction.getConnection();
81      transaction.commit();
82      transaction.close();
83      assertThat(connection.getNumberCommits()).as("should not call commit on a Connection with autocommit").isEqualTo(0);
84      assertThat(connection.isClosed()).as("should close the Connection").isTrue();
85    }
86  
87    @Test
88    void shouldIgnoreAutocommit() throws Exception {
89      SpringManagedTransactionFactory transactionFactory = new SpringManagedTransactionFactory();
90      SpringManagedTransaction transaction = (SpringManagedTransaction) transactionFactory.newTransaction(dataSource,
91          null, true);
92      transaction.getConnection();
93      transaction.commit();
94      transaction.close();
95      assertThat(connection.getNumberCommits()).as("should call commit on Connection").isEqualTo(1);
96      assertThat(connection.isClosed()).as("should close the Connection").isTrue();
97    }
