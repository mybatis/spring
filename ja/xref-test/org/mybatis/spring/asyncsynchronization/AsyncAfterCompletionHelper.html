16  package org.mybatis.spring.asyncsynchronization;
17  
18  import java.lang.reflect.InvocationHandler;
19  import java.lang.reflect.InvocationTargetException;
20  import java.lang.reflect.Method;
21  import java.lang.reflect.Proxy;
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  import org.springframework.transaction.support.TransactionSynchronization;
26  
27  /**
28   * For use as ByteMan helper
29   *
30   * @author Alex Rykov
31   */
32  @SuppressWarnings("unused")
33  public class AsyncAfterCompletionHelper {
34    /**
35     * Invocation handler that performs afterCompletion on a separate thread See Github issue #18
36     *
37     * @author Alex Rykov
38     */
39    static class AsyncAfterCompletionInvocationHandler implements InvocationHandler {
40  
41      private Object target;
42  
43      AsyncAfterCompletionInvocationHandler(Object target) {
44        this.target = target;
45      }
46  
47      @Override
48      public Object invoke(final Object proxy, final Method method, final Object[] args) throws Throwable {
49        if (!"afterCompletion".equals(method.getName())) {
50          return method.invoke(target, args);
51        }
52        final Set<Object> retValSet = new HashSet<>();
53        final Set<Throwable> exceptionSet = new HashSet<>();
54        var thread = new Thread(() -> {
55          try {
56            retValSet.add(method.invoke(target, args));
57          } catch (InvocationTargetException ite) {
58            exceptionSet.add(ite.getCause());
59  
60          } catch (IllegalArgumentException | IllegalAccessException e) {
61            exceptionSet.add(e);
62  
63          }
64        });
65        thread.start();
66        thread.join();
67        if (exceptionSet.isEmpty()) {
68          return retValSet.iterator().next();
69        }
70        throw exceptionSet.iterator().next();
71      }
72  
73    }
74  
75    /**
76     * Creates proxy that performs afterCompletion call on a separate thread
77     */
78    public TransactionSynchronization createSynchronizationWithAsyncAfterComplete(
79        TransactionSynchronization synchronization) {
80      if (Proxy.isProxyClass(synchronization.getClass())
81          && Proxy.getInvocationHandler(synchronization) instanceof AsyncAfterCompletionInvocationHandler) {
82        // avoiding double wrapping just in case
83        return synchronization;
84      }
85      Class<?>[] interfaces = { TransactionSynchronization.class };
86      return (TransactionSynchronization) Proxy.newProxyInstance(synchronization.getClass().getClassLoader(), interfaces,
87          new AsyncAfterCompletionInvocationHandler(synchronization));
88    }
89  
90  }
